# Harder task: balanced binary tree (AVL insert)

Each snippet implements the **same algorithm**: **AVL insertion** (integer keys) with **LL / RR / LR / RL** rotations when balance factors exceed ±1. No delete. Demo inserts `10, 20, 30, 40, 50, 25`.

For the lighter “sum of even squares” exercise, see **[multi-lang & token counts](/en/appendix/multi-lang-token-compare)**.

## Token methodology

- Encoder **`cl100k_base`** via `gpt-tokenizer`; counts **only the code inside each fence**.  
- Strings must stay **identical** to `web/scripts/count-appendix-balanced-tree.mjs`. After edits run `pnpm run tokens:appendix:avl` and refresh the tables.  
- **Not** a performance or safety review (the C++ sample uses raw `new` and omits cleanup on purpose).

## JavaScript (394 tokens)

```javascript
class AVLNode {
  constructor(key) {
    this.key = key;
    this.left = this.right = null;
    this.h = 1;
  }
}

function gh(n) {
  return n ? n.h : 0;
}

function upd(n) {
  n.h = 1 + Math.max(gh(n.left), gh(n.right));
}

function bal(n) {
  return n ? gh(n.left) - gh(n.right) : 0;
}

function rotR(y) {
  const x = y.left;
  y.left = x.right;
  x.right = y;
  upd(y);
  upd(x);
  return x;
}

function rotL(x) {
  const y = x.right;
  x.right = y.left;
  y.left = x;
  upd(x);
  upd(y);
  return y;
}

function avlInsert(root, key) {
  if (!root) return new AVLNode(key);
  if (key < root.key) root.left = avlInsert(root.left, key);
  else if (key > root.key) root.right = avlInsert(root.right, key);
  else return root;
  upd(root);
  const b = bal(root);
  if (b > 1 && key < root.left.key) return rotR(root);
  if (b > 1 && key > root.left.key) {
    root.left = rotL(root.left);
    return rotR(root);
  }
  if (b < -1 && key > root.right.key) return rotL(root);
  if (b < -1 && key < root.right.key) {
    root.right = rotR(root.right);
    return rotL(root);
  }
  return root;
}

let r = null;
for (const k of [10, 20, 30, 40, 50, 25]) r = avlInsert(r, k);
```

class AVLNode:
    __slots__ = ("key", "left", "right", "h")

    def __init__(self, key: int):
        self.key = key
        self.left = self.right = None
        self.h = 1


def _gh(n: AVLNode | None) -> int:
    return n.h if n else 0


def _upd(n: AVLNode) -> None:
    n.h = 1 + max(_gh(n.left), _gh(n.right))


def _bal(n: AVLNode) -> int:
    return _gh(n.left) - _gh(n.right)


def _rot_r(y: AVLNode) -> AVLNode:
    x = y.left
    assert x is not None
    y.left, x.right = x.right, y
    _upd(y)
    _upd(x)
    return x


def _rot_l(x: AVLNode) -> AVLNode:
    y = x.right
    assert y is not None
    x.right, y.left = y.left, x
    _upd(x)
    _upd(y)
    return y


def avl_insert(root: AVLNode | None, key: int) -> AVLNode:
    if root is None:
        return AVLNode(key)
    if key < root.key:
        root.left = avl_insert(root.left, key)
    elif key > root.key:
        root.right = avl_insert(root.right, key)
    else:
        return root

    bf = _bal(root)
    if bf > 1 and root.left and key < root.left.key:

        return _rot_l(root)

## Rust (578 tokens)

```rust
#[derive(Debug)]
struct Node {
    key: i32,
    left: Link,
    right: Link,
    h: i8,
}

type Link = Option<Box<Node>>;

impl Node {
    fn new(key: i32) -> Self {
        Node {
            key,
            left: None,
            right: None,
            h: 1,
        }
    }
}

fn gh(n: &Link) -> i8 {
    n.as_ref().map_or(0, |b| b.h)
}

fn upd(n: &mut Node) {
    n.h = 1 + gh(&n.left).max(gh(&n.right));
}

fn bal(n: &Node) -> i8 {
    gh(&n.left) - gh(&n.right)
}

fn rot_r(mut y: Box<Node>) -> Box<Node> {
    let mut x = y.left.take().unwrap();
    y.left = x.right.take();
    x.right = Some(y);
    if let Some(ref mut yr) = x.right {
        upd(yr);
    }
    upd(&mut x);
    x
}

fn rot_l(mut x: Box<Node>) -> Box<Node> {
    let mut y = x.right.take().unwrap();
    x.right = y.left.take();
    y.left = Some(x);
    if let Some(ref mut yl) = y.left {
        upd(yl);
    }
    upd(&mut y);
    y
}

fn insert(t: Link, key: i32) -> Link {
    let mut n = match t {
        None => return Some(Box::new(Node::new(key))),
        Some(b) => b,
    };
    if key < n.key {
        n.left = insert(n.left.take(), key);
    } else if key > n.key {
        n.right = insert(n.right.take(), key);
    } else {
        return Some(n);
    }
    upd(&mut n);
    let b = bal(&n);
    if b > 1 {
        let lk = n.left.as_ref().unwrap().key;
        if key < lk {
            return Some(rot_r(n));
        }
        n.left = Some(rot_l(n.left.take().unwrap()));
        return Some(rot_r(n));
    }
    if b < -1 {
        let rk = n.right.as_ref().unwrap().key;
        if key > rk {
            return Some(rot_l(n));
        }
        n.right = Some(rot_r(n.right.take().unwrap()));
        return Some(rot_l(n));
    }
    Some(n)
}

fn main() {
    let mut r: Link = None;
    for k in [10, 20, 30, 40, 50, 25] {
        r = insert(r, k);
    }
    let _ = r;
}
```

## Takeaways

- **JavaScript / Ruby** stay cheapest for this style of AVL; **Rust** pays the usual **`Option<Box>` tax** in tokens.  
- **Python** annotations + `assert` noticeably inflate counts—strip them if you only care about runtime behavior.  
- Compare against the **[simple task page](/en/appendix/multi-lang-token-compare)** to see how complexity explodes token use (roughly **6–15×** vs the even-squares micro-sample).
