## 共轭梯度求解

```python
def conjugate_gradient(H, b, n_iter=10):
    # 求解 Hx = b
    x = torch.zeros_like(b)
    r = b.clone()
    p = b.clone()
    
    for _ in range(n_iter):
        Ap = H(p)
        alpha = (r @ r) / (p @ Ap + 1e-8)
        x = x + alpha * p
        r_new = r - alpha * Ap
        if r_new.norm() < 1e-8:
            break
        beta = (r_new @ r_new) / (r @ r)
        p = r_new + beta * p
        r = r_new
    
    return x
```
