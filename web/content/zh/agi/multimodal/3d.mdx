## 3D 稀疏卷积

```python
class SparseConv3d(nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size=3):
        super().__init__()
        self.conv = nn.Conv3d(
            in_channels, out_channels, 
            kernel_size,
            padding=kernel_size//2,
            bias=False
        )
    
    def forward(self, x, coords):
        """
        x: 特征 (N, C)
        coords: 坐标 (N, 3)
        """
        # 构建 3D 网格
        voxel_grid = self.scatter(x, coords)  # 哈希表查询
        
        # 在密集网格上卷积
        out = self.conv(voxel_grid)
        
        # 聚合回点
        return out, coords
```

## 哈希表 + 规则本

```python
# 3D 点 → 哈希表 → 特征查询
# 实现高效的稀疏操作

def hash_query(coords, resolution, hash_size):
    # 多尺度哈希
    for scale in resolutions:
        scaled = coords // scale
        hash_key = (scaled.x * 73856093 ^ scaled.y * 19349663 ^ scaled.z * 83492791) % hash_size
        yield hash_key, scale
```
