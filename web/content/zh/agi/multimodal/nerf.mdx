# NeRF 神经辐射场

通过神经网络表示3D场景。

---

## 核心思想

```python
class NeRF(nn.Module):
    def __init__(self):
        self.pos_embed = PosEmbedding(3)  # 位置编码
        self.dir_embed = PosEmbedding(3)
        self.mlp = nn.Sequential(
            nn.Linear(60, 256), nn.ReLU(),
            nn.Linear(256, 256), nn.ReLU(),
            nn.Linear(256, 4)  # RGB + density
        )
    
    def forward(self, x, d):
        # x: 3D 位置, d: 视角方向
        x = self.pos_embed(x)
        d = self.dir_embed(d)
        h = self.mlp(x)
        rgb, sigma = h[..., :3], h[..., 3]
        return rgb, sigma
```

## 体素渲染

```python
def render(rgb, sigma, depths):
    # 体素渲染
    deltas = depths[1:] - depths[:-1]
    alphas = 1 - torch.exp(-sigma * deltas)
    
    # 累积
    transmittance = torch.cumprod(1 - alphas, dim=-1)
    weights = transmittance * alphas
    
    rgb_final = (weights.unsqueeze(-1) * rgb).sum(dim=1)
    return rgb_final
```
